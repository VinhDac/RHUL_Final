# --- Loss Function (from Week3Exercise1) ---
loss_function = nn.CrossEntropyLoss()

# --- Accuracy and Loss (from Week3Exercise2, adapted for multiclass using Week3Exercise1's torch.max) ---
def accuracy_and_loss(net, loss_function, dataloader):
    total_correct = 0
    total_loss = 0.0
    total_examples = 0
    n_batches = 0
    with torch.no_grad():
        for data in dataloader:
            images, labels = data
            outputs = net(images)
            batch_loss = loss_function(outputs, labels)
            n_batches += 1
            total_loss += batch_loss.item()
            _, predicted = torch.max(outputs, dim=1)
            total_correct += (predicted == labels).sum().item()
            total_examples += labels.size(0)
    accuracy = total_correct / total_examples
    mean_loss = total_loss / n_batches
    return (accuracy, mean_loss)

# --- Training function (from Week3Exercise2: define_and_train, adapted for multiclass) ---
def define_and_train(NN_class, n_epochs, training_set, test_set, batch_size=32, weight_decay=0.0):
    trainloader = torch.utils.data.DataLoader(training_set, batch_size=batch_size, shuffle=True)
    testloader = torch.utils.data.DataLoader(test_set, batch_size=batch_size, shuffle=True)

    thenet = NN_class()
    optimizer1 = optim.Adam(thenet.parameters(), weight_decay=weight_decay)

    train_acc = []
    val_acc = []
    train_loss = []
    val_loss = []

    for epoch in range(n_epochs):
        total_loss = 0
        total_correct = 0
        total_examples = 0
        n_mini_batches = 0

        for i, mini_batch in enumerate(trainloader, 0):
            images, labels = mini_batch
            optimizer1.zero_grad()
            outputs = thenet(images)
            loss = loss_function(outputs, labels)
            loss.backward()
            optimizer1.step()

            n_mini_batches += 1
            total_loss += loss.item()
            _, predicted = torch.max(outputs, dim=1)
            total_correct += (predicted == labels).sum().item()
            total_examples += labels.size(0)

        epoch_training_accuracy = total_correct / total_examples
        epoch_training_loss = total_loss / n_mini_batches
        epoch_val_accuracy, epoch_val_loss = accuracy_and_loss(thenet, loss_function, testloader)

        if (epoch + 1) % 10 == 0:
            print('Epoch %d loss: %.3f acc: %.3f val_loss: %.3f val_acc: %.3f'
                  % (epoch + 1, epoch_training_loss, epoch_training_accuracy, epoch_val_loss, epoch_val_accuracy))

        train_loss.append(epoch_training_loss)
        train_acc.append(epoch_training_accuracy)
        val_loss.append(epoch_val_loss)
        val_acc.append(epoch_val_accuracy)

    history = {'train_loss': train_loss, 'train_acc': train_acc, 'val_loss': val_loss, 'val_acc': val_acc}
    return (history, thenet)

# Architecture from Week3Exercise2's NN_two_hidden, adapted for MNIST:
# - Input: 784 (28x28 flattened) instead of 64
# - Output: 10 classes instead of 1 (removed Sigmoid, using CrossEntropyLoss)
# - Larger hidden layers (256, 128) to ensure overfitting capacity
# ====================================================================
class FC_Model(nn.Module):
    def __init__(self):
        super(FC_Model, self).__init__()
        self.layers = nn.Sequential(
            nn.Flatten(),
            nn.Linear(784, 256),
            nn.ReLU(),
            nn.Linear(256, 128),
            nn.ReLU(),
            nn.Linear(128, 10))

    def forward(self, x):
        x = self.layers(x)
        return x

# --- Fit power law: error ~ n^(-alpha) ---
# Taking log: log(error) = -alpha * log(n) + c
log_sizes = np.log(np.array(fc_lc_sizes))
log_errors = np.log(np.array(fc_lc_errors))
coeffs_fc = np.polyfit(log_sizes, log_errors, 1)
alpha_fc = -coeffs_fc[0]

# --- CNN Model (from MNIST_Siamese: embedding_model_2 architecture) ---
# Conv2d(1,20,3) -> ReLU -> Conv2d(20,40,3) -> ReLU -> MaxPool2d(2,2) ->
# Conv2d(40,40,3) -> ReLU -> MaxPool2d(2,2) -> Flatten -> Linear(1000,128) -> ReLU -> Linear(128,10)
class CNN_Model(nn.Module):
    def __init__(self):
        super(CNN_Model, self).__init__()
        self.layers = nn.Sequential(
            nn.Conv2d(1, 20, 3),       # 1x28x28 -> 20x26x26
            nn.ReLU(),
            nn.Conv2d(20, 40, 3),      # 20x26x26 -> 40x24x24
            nn.ReLU(),
            nn.MaxPool2d(2, 2),        # 40x24x24 -> 40x12x12
            nn.Conv2d(40, 40, 3),      # 40x12x12 -> 40x10x10
            nn.ReLU(),
            nn.MaxPool2d(2, 2),        # 40x10x10 -> 40x5x5 = 1000
            nn.Flatten(),
            nn.Linear(1000, 128),
            nn.ReLU(),
            nn.Linear(128, 10))

    def forward(self, x):
        x = self.layers(x)
        return x
